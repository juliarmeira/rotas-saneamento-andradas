        // Focado inicialmente na região central de Andradas

        }).addTo(map);

        let controleRota = null;

        // --- 3. LÓGICA DE LEITURA DO ARQUIVO KML ---
        document.getElementById('kmlFile').addEventListener('change', function(e) {
            const file = e.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            
            reader.onload = function(event) {
                const kmlText = event.target.result;
                const parser = new DOMParser();
                const kmlDoc = parser.parseFromString(kmlText, "text/xml");
                
                // Extrair as coordenadas dos pontos (Placemarks)
                const placemarks = kmlDoc.getElementsByTagName("Placemark");
                const waypoints = [];

                for (let i = 0; i < placemarks.length; i++) {
                    const coordString = placemarks[i].getElementsByTagName("coordinates")[0]?.textContent;
                    if (coordString) {
                        // KML salva como: longitude, latitude, altitude
                        const coords = coordString.trim().split(',');
                        const lng = parseFloat(coords[0]);
                        const lat = parseFloat(coords[1]);
                        waypoints.push(L.latLng(lat, lng));
                    }
                }

                if (waypoints.length > 0) {
                    tracarRota(waypoints);
                } else {
                    alert("Nenhuma coordenada válida encontrada no arquivo KML.");
                }
            };
            
            reader.readAsText(file);
        });

        // --- 4. TRAÇAR ROTA E CRIAR POPUPS ---
        function tracarRota(waypoints) {
            // Se já existir uma rota, remove a antiga antes de criar a nova
            if (controleRota) {
                map.removeControl(controleRota);
            }

            controleRota = L.Routing.control({
                waypoints: waypoints,
                routeWhileDragging: false,
                addWaypoints: false,
                fitSelectedRoutes: true,
                show: false, // Esconde o painel nativo
                createMarker: function(i, waypoint, n) {
                    // Pega o dado técnico correspondente à ordem do ponto no KML
                    // Se o KML tiver mais pontos que o nosso JSON, usa dados genéricos

                        nome: `Ponto Desconhecido ${i+1}`, tipo: "A verificar", status: "-", capacidade: "-", 
                        pendencias: "Ponto extra no arquivo KML sem dados correspondentes.", diretrizPlano: "-" 
                    };

                    // Define estilo da tag (Prefeitura fica verde, ETEs amarelo)
                    const badgeClass = dado.id === "origem" ? "badge institucional" : "badge";

                    // Monta o layout HTML de dentro da caixinha (Popup)
                    const popupContent = `
                        <div class="custom-popup">
                            <h4>${dado.nome}</h4>
                            <span class="${badgeClass}">${dado.tipo} | ${dado.status}</span>
                            <p><strong>Capacidade:</strong> ${dado.capacidade}</p>
                            <div class="pendencias">

                                ${dado.diretrizPlano}
                            </div>
                        </div>
                    `;

                    // Retorna o marcador azul padrão do Leaflet com a nossa caixinha atrelada
                    return L.marker(waypoint.latLng).bindPopup(popupContent);
                }
            }).addTo(map);
        }
